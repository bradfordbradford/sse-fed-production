    Render Graph

    For more information on Unix Encoded Timestamps: Unix Time Stamps, Epoch Converter.

    <!-- Graph Data (x: --Unix Encoded Timestamp--) -->

<script>
	var data = [ 
		{ x: 1368421200, y: 5.5 },
		{ x: 1370062800, y: 260 },
		{ x: 1373432400, y: 120 },
		{ x: 1376456400, y: 200 },
		{ x: 1379739600, y: 40 },
		{ x: 1380603600, y: 140 },
	];

	// Import Color Palette
	// var palette = new Rickshaw.Color.Palette();

	// Graph Specs
	var graph = new Rickshaw.Graph( {
		element: document.getElementById("chart"),
		height: 300,
		width: 560,
		series: [
			{
				// name: "SMU",
				data: data,
				color: "#6f992d"
				// palette.color()
			}
		
		]
	} );

	// Display Y Axis
	var yAxis = new Rickshaw.Graph.Axis.Y( {
	    graph: graph,
	    orientation: 'left',
	    element: document.getElementById('y_axis'),
	} );

	// Render Y Axis
	yAxis.render();

	// Populate Tooltip with Custom Data
	var hoverDetail = new Rickshaw.Graph.HoverDetail( {
		graph: graph,
		formatter: function(series, x, y) {
			var date = '' + new Date(x * 1000).toUTCString() + '';
			var swatch = '';
			// var content = swatch + date + '
    ' + '' + series.name + ": $" + (y) + '' ;
			// without series name:
			var content = swatch + date + '
    ' + '' + "$" + (y) + '' ;
			return content;
		}
	} );

	// Render Graph
	graph.render();
</script>
