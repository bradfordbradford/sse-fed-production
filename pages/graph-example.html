// Chart Script
<script>

// Graph Data
var data = [ { x: 1367916284, y: 330.5 }, { x: 1368009884, y: 49.89 }, { x: 1368009884, y: 38.40 }, { x: 1368121484, y: 20.66 }, { x: 1368197094, y: 16 } ];

// Import Color Palette
var palette = new Rickshaw.Color.Palette();

// Graph Specs
var graph = new Rickshaw.Graph( {
	element: document.getElementById("chart"),
	height: 300,
	width: 540,
	series: [
		{
			name: "SMU",
			data: data,
			color: palette.color()
		}
	
	]
} );

// Display Y Axis
var yAxis = new Rickshaw.Graph.Axis.Y( {
    graph: graph,
    orientation: 'left',
    element: document.getElementById('y_axis'),
} );

// Render Y Axis
yAxis.render();

// Populate Tooltip with Custom Data
var hoverDetail = new Rickshaw.Graph.HoverDetail( {
	graph: graph,
	formatter: function(series, x, y) {
		var date = '' + new Date(x * 1000).toUTCString() + '';
		var swatch = '';
		var content = swatch + series.name + ": $" + (y) + '
    ' + date;
		return content;
	}
} );

// Render Graph
graph.render();
</script>
